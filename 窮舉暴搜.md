#include <bits/stdc++.h>
using namespace std;

double t(int n, int s[])
{
  int p = 3; //人數
  int time[p] = {40, 44, 60}; //每人秒數
  double ans = 0; //總秒數

  for (int j = 0; j < n; j++)
    for (int i = 0; i < s[j]; i++)
      ans = ans + time[j%p] * pow(1.1, i) * pow(1.2, j/3);

  return(ans);
}

void print(vector<int>& v, int level)
{
  int tmp[10005];

  for(int i = 0; i <= level; i++) tmp[i] = v[i];

  sort(tmp, tmp + level+1);
  do //列出所有加法(依字典排序)
  {
    if(level + 1 >= 3) //每個人都要跑
    {
      cout << level+1 << ","; //io redirect輸出成.csv
      for(int i = 0; i <= level; i++)
        cout << tmp[i] << " ";

      cout << ",";
      cout << t(level+1, tmp);

      cout << endl;
    }
  } while (next_permutation(tmp, tmp + level+1));
}

void part(int n, vector<int>& v, int level)
{
  int first;

  if(n < 1) return;

  v[level] = n;
  print(v, level);

  if(level == 0)
    first = 1; //第一次從1開始分割
  else
    first = v[level-1]; //從上次處理完的地方繼續

  for(int i = first; i <= n/2; i++)
  {
    v[level] = i;
    part(n-i, v, level+1); //繼續處理還沒分割過的部分
  }
}

int main()
{
  ios::sync_with_stdio(0);
  cin.tie(0); cout.tie(0);

  int num;
  vector<int> v(num);
  cin >> num;

  part(num, v, 0);

  return 0;
}
